154inline Vector3 Matrix4x4::translation() const noexcept
155{
156 const DirectX::XMVECTOR& row3 = _matrix.r[3];
157 const DirectX::XMVECTOR wVec = DirectX::XMVectorSplatW(row3);
158
159 const DirectX::XMVECTOR zeroMask = DirectX::XMVectorEqual(wVec, DirectX::XMVectorZero());
160
161 const DirectX::XMVECTOR safeDivided = DirectX::XMVectorSelect(
162 DirectX::XMVectorDivide(row3, wVec),
163 DirectX::XMVectorZero(),
164 zeroMask
165 );
166
167 return Vector3(safeDivided);
168}

218inline bool Matrix4x4::inverse3DAffine(const Matrix4x4& input, Matrix4x4& output) noexcept
219{
220 using namespace DirectX;
221
222 XMVECTOR scale, rotationQuat, translation;
223 if (!XMMatrixDecompose(&scale, &rotationQuat, &translation, input._matrix))
224 return false;
225
226 if (XMVector4NearEqual(scale, XMVectorZero(), XMVectorReplicate(1e-6f)))
227 return false;
228
229 XMMATRIX invTranslate = XMMatrixTranslationFromVector(-translation);
230 XMVECTOR invRotation = XMQuaternionInverse(rotationQuat);
231 XMMATRIX invScale = XMMatrixScalingFromVector(XMVectorReciprocal(scale));
232
233 output._matrix = invTranslate* XMMatrixRotationQuaternion(invRotation) * invScale;
234
235 return true;
236}

248inline Quaternion Matrix4x4::rotation() const noexcept
249{
250 auto scale = this->scale();
251 if (DirectX::XMVector4NearEqual(scale, DirectX::XMVectorZero(),
252 DirectX::XMVectorReplicate(0.001f))) {
253 return Quaternion::identity;
254 }
255
256 DirectX::XMVECTOR v0 = DirectX::XMVectorDivide(_matrix.r[0], scale);
257 DirectX::XMVECTOR v1 = DirectX::XMVectorDivide(_matrix.r[1], scale);
258 DirectX::XMVECTOR v2 = DirectX::XMVectorDivide(_matrix.r[2], scale);
259
260 DirectX::XMMATRIX rotMatrix = { v0, v1, v2, _matrix.r[3] };
261
262 return Quaternion(DirectX::XMQuaternionRotationMatrix(rotMatrix));
263}

303inline bool Matrix4x4::isValidTRS() const noexcept
304{
305 static const DirectX::XMVECTORF32 lastRowCheck = { 0,0,0,1 };
306 if (!DirectX::XMVector4Equal(getColumn(3), lastRowCheck))
